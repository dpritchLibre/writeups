% ------------------------------------------------------------------------------
% configure the appearance of the section titles.  Requires the following
% packages:
%
% \usepackage{titlesec}
% ------------------------------------------------------------------------------

% titlesec documentation:
% http://ctan.mackichan.com/macros/latex/contrib/titlesec/titlesec.pdf

% see pgs 4-5 of the documentation for the `titlespacing` syntax.  The command
% signature is:
% \titlespacing*{<command>}{<left>}{<before-sep>}{<after-sep>}[<right-sep>]
\titlespacing*{\section}{0pt}{10mm plus 1ex minus .2ex}{2mm plus .2ex}
\titlespacing*{\subsection}{0pt}{7mm plus 1ex minus .2ex}{2mm plus .2ex}
\titlespacing*{\subsubsection}{0pt}{5mm plus 1ex minus .2ex}{2mm plus .2ex}
